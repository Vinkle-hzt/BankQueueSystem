70 static void md5(const uint8_t *initial_msg, size_t initial_len, uint8_t *digest)
71 {
72  // These vars will contain the hash
73  uint32_t h0, h1, h2, h3;
74 
75  // Message (to prepare)
76  uint8_t *msg = NULL;
77 
78  size_t new_len, offset;
79  uint32_t w[16];
80  uint32_t a, b, c, d, i, f, g, temp;
81 
82  // Initialize variables - simple count in nibbles:
83  h0 = 0x67452301;
84  h1 = 0xefcdab89;
85  h2 = 0x98badcfe;
86  h3 = 0x10325476;
87 
88  //Pre-processing:
89  //append "1" bit to message
90  //append "0" bits until message length in bits ≡ 448 (mod 512)
91  //append length mod (2^64) to message
92 
93  for (new_len = initial_len + 1; new_len % (512 / 8) != 448 / 8; new_len++)
94  ;
95 
96  msg = (uint8_t *)malloc(new_len + 8);
97  memcpy(msg, initial_msg, initial_len);
98  msg[initial_len] = 0x80; // append the "1" bit; most significant bit is "first"
99  for (offset = initial_len + 1; offset < new_len; offset++)
100  msg[offset] = 0; // append "0" bits
101 
102  // append the len in bits at the end of the buffer.
103  to_bytes(initial_len * 8, msg + new_len);
104  // initial_len>>29 == initial_len*8>>32, but avoids overflow.
105  to_bytes(initial_len >> 29, msg + new_len + 4);
106 
107  // Process the message in successive 512-bit chunks:
108  // for each 512-bit chunk of message:
109  for (offset = 0; offset < new_len; offset += (512 / 8))
110  {
111 
112  // break chunk into sixteen 32-bit words w[j], 0 ≤ j ≤ 15
113  for (i = 0; i < 16; i++)
114  w[i] = to_int32(msg + offset + i * 4);
115 
116  // Initialize hash value for this chunk:
117  a = h0;
118  b = h1;
119  c = h2;
120  d = h3;
121 
122  // Main loop:
123  for (i = 0; i < 64; i++)
124  {
125 
126  if (i < 16)
127  {
128  f = (b & c) | ((~b) & d);
129  g = i;
130  }
131  else if (i < 32)
132  {
133  f = (d & b) | ((~d) & c);
134  g = (5 * i + 1) % 16;
135  }
136  else if (i < 48)
137  {
138  f = b ^ c ^ d;
139  g = (3 * i + 5) % 16;
140  }
141  else
142  {
143  f = c ^ (b | (~d));
144  g = (7 * i) % 16;
145  }
146 
147  temp = d;
148  d = c;
149  c = b;
150  b = b + LEFTROTATE((a + f + k[i] + w[g]), r[i]);
151  a = temp;
152  }
153 
154  // Add this chunk's hash to result so far:
155  h0 += a;
156  h1 += b;
157  h2 += c;
158  h3 += d;
159  }
160 
161  // cleanup
162  free(msg);
163 
164  // var char digest[16] := h0 append h1 append h2 append h3 //(Output is in little-endian)
165  to_bytes(h0, digest);
166  to_bytes(h1, digest + 4);
167  to_bytes(h2, digest + 8);
168  to_bytes(h3, digest + 12);
169 }
